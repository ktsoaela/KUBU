Basic Template

A simple template to get you started.

  <!DOCTYPE html>
  <html lang="en">
  <head>
      <meta charset="UTF-8">
      <meta name="viewport" content="width=device-width, initial-scale=1.0">
      <title>My Kubu Project</title>
      <link rel="stylesheet" href="kubu.css">
      <script src="htmx.min.js"></script>
  </head>
  <body>
      <header>
          <h1>Welcome to My Kubu Project</h1>
      </header>
      <main>
          <section>
              <h2>Main Content</h2>
              <p>This is where your content goes.</p>
          </section>
      </main>
  </body>
  </html>
